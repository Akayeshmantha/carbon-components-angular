class ModalContainer {
	destroy: Function;
	closeModal(): void {
		this.destroy();
	}
}

/**
 * Applying the decorator metadata to the target class (`Modal`).
 * @export
 * @returns {Object}
 */
export default function Modal() {
	return function(target) {
		Object.assign(target.prototype, ModalContainer.prototype);
	};
}
